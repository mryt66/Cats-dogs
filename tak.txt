import matplotlib.pyplot as plt
import numpy as np
from scipy.io import wavfile
from scipy.fft import fft, ifft
import soundfile as sf

%matplotlib inline
plt.rcParams['figure.dpi'] = 100
plt.rcParams['figure.figsize'] = (9, 7)
data, samplerate = sf.read('cat_01.wav', dtype='float32')
x=np.array(data)
y=fft(x)

#wykres
plt.subplot(2,1,1)
plt.plot(y,'r')
plt.xlabel("FFT")
plt.tight_layout()
plt.show()
